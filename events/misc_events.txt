namespace = dragon_misc

country_event = {
	id = dragon_misc.360
	hide_window = yes
	#is_triggered_only = yes
	
	trigger = {
		NOT = {has_country_flag = is_under_war}
		any_war = {
			any_defender = { is_same_value = this }
		}
	}

	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		set_country_flag = is_under_war
	}
}
